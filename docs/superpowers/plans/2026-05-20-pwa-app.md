# PWA App — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the Plannen web app into an installable PWA with offline shell, camera-first photo capture, native share-sheet, Android share-target, and Web Push notifications.

**Architecture:** Use `vite-plugin-pwa` (Workbox under the hood) to generate the service worker and manifest from a single config. Push uses VAPID + `web-push` on the Hono backend, with subscriptions stored in a new `push_subscriptions` table. iOS push requires the user to install the PWA first (16.4+) — Android push works once the user accepts the permission prompt. No Capacitor, no app store, no rewrite.

**Tech Stack:** Vite 5, React 18, vite-plugin-pwa, Workbox, web-push (Node), Hono, Postgres.

---

## Scope (locked with user 2026-05-20)

1. **Installable + offline shell** — manifest, icons, iOS meta tags, Workbox SW with offline shell + runtime-cached API GETs.
2. **Camera-first photo capture** — `capture="environment"` on EventForm photo extract input.
3. **Native share-sheet OUT** — `navigator.share` helper used by share modals (EventShareModal, StoryShareModal). Falls back to existing `wa.me` link on desktop.
4. **Web Share Target (Android)** — manifest `share_target` entry + `/share` route that hands the shared payload to EventForm pre-filled.
5. **Web Push** — VAPID keys via env, `push_subscriptions` table, subscribe/unsubscribe/test endpoints, SW push handler, Settings toggle + test button. One real trigger wired: when a watch finds a change, push a notification.

**Out of scope for this PR:** scheduled push triggers (RSVP day-of reminders, daily briefing push). They need a job scheduler — follow-up.

```ts
/// <reference lib="webworker" />
import { precacheAndRoute, cleanupOutdatedCaches } from 'workbox-precaching'
import { registerRoute, NavigationRoute } from 'workbox-routing'
import { NetworkFirst, StaleWhileRevalidate } from 'workbox-strategies'
import { ExpirationPlugin } from 'workbox-expiration'

declare const self: ServiceWorkerGlobalScope

cleanupOutdatedCaches()
precacheAndRoute(self.__WB_MANIFEST)

// App shell: SPA navigation — serve precached index.html offline
registerRoute(
  new NavigationRoute(
    new NetworkFirst({
      cacheName: 'plannen-shell',
      networkTimeoutSeconds: 3,
    }),
    { denylist: [/^\/api\//, /^\/share/, /^\/auth\//] },
  ),
)

// GET API responses: stale-while-revalidate for fast UI, 1-day cap
registerRoute(
  ({ request, url }) => request.method === 'GET' && url.pathname.startsWith('/api/'),
  new StaleWhileRevalidate({
    cacheName: 'plannen-api',
    plugins: [new ExpirationPlugin({ maxAgeSeconds: 24 * 60 * 60, maxEntries: 200 })],
  }),
)

// Push event: show a notification
self.addEventListener('push', (event) => {
  const data = (() => {
    try {
      return event.data?.json() ?? {}
    } catch {
      return { title: 'Plannen', body: event.data?.text() ?? '' }
    }
  })()
  const title = data.title ?? 'Plannen'
  const body = data.body ?? ''
  const url = data.url ?? '/dashboard'
  event.waitUntil(
    self.registration.showNotification(title, {
      body,
      icon: '/icons/icon-192.png',
      badge: '/icons/icon-192.png',
      data: { url },
      tag: data.tag ?? 'plannen',
    }),
  )
})

self.addEventListener('notificationclick', (event) => {
  event.notification.close()
  const url = (event.notification.data as { url?: string } | undefined)?.url ?? '/dashboard'
  event.waitUntil(
    self.clients.matchAll({ type: 'window' }).then((clients) => {
      for (const client of clients) {
        if ('focus' in client) {
          client.navigate(url)
          return (client as WindowClient).focus()
        }
      }
      return self.clients.openWindow(url)
    }),
  )
})

self.addEventListener('message', (event) => {
  if (event.data?.type === 'SKIP_WAITING') self.skipWaiting()
})
```

```ts
export interface SharePayload {
  title?: string
  text?: string
  url?: string
}

/**
 * Try the native share-sheet. Returns true if the share completed (or the user
 * cancelled it), false if the platform doesn't support Web Share.
 */
export async function shareNative(payload: SharePayload): Promise<boolean> {
  if (typeof navigator === 'undefined' || typeof navigator.share !== 'function') {
    return false
  }
  try {
    await navigator.share(payload)
    return true
  } catch (err) {
    // AbortError = user cancelled; still counts as "we tried natively"
    if ((err as DOMException)?.name === 'AbortError') return true
    return false
  }
}
```

  }, [params, navigate])

  return (

  )

-- Web Push subscriptions per user. One row per browser/device.
create table if not exists public.push_subscriptions (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  endpoint text not null,
  p256dh text not null,
  auth text not null,
  user_agent text,
  created_at timestamptz not null default now(),
  last_used_at timestamptz,
  unique (user_id, endpoint)
);

create index if not exists push_subscriptions_user_id_idx on public.push_subscriptions (user_id);

alter table public.push_subscriptions enable row level security;

create policy "Users select their own push subscriptions"
  on public.push_subscriptions for select
  using (auth.uid() = user_id);

create policy "Users insert their own push subscriptions"
  on public.push_subscriptions for insert
  with check (auth.uid() = user_id);

create policy "Users delete their own push subscriptions"
  on public.push_subscriptions for delete
  using (auth.uid() = user_id);

create policy "Users update their own push subscriptions"
  on public.push_subscriptions for update
  using (auth.uid() = user_id)
  with check (auth.uid() = user_id);

```ts
import webpush from 'web-push'
import { pool } from '../db.js'

const PUBLIC = process.env.VAPID_PUBLIC_KEY
const PRIVATE = process.env.VAPID_PRIVATE_KEY
const SUBJECT = process.env.VAPID_SUBJECT ?? 'mailto:hello@plannen.app'

let configured = false
function ensureConfigured() {
  if (configured) return true
  if (!PUBLIC || !PRIVATE) return false
  webpush.setVapidDetails(SUBJECT, PUBLIC, PRIVATE)
  configured = true
  return true
}

export interface PushPayload {
  title: string
  body?: string
  url?: string
  tag?: string
}

export interface PushResult {
  attempted: number
  sent: number
  removed: number
  errors: string[]
}

/**
 * Send a push to every active subscription for the user. Removes dead
 * subscriptions (404/410). Returns a delivery summary.
 */
export async function sendPush(userId: string, payload: PushPayload): Promise<PushResult> {
  const result: PushResult = { attempted: 0, sent: 0, removed: 0, errors: [] }
  if (!ensureConfigured()) {
    result.errors.push('VAPID keys not configured')
    return result
  }
  const { rows } = await pool.query<{ id: string; endpoint: string; p256dh: string; auth: string }>(
    'select id, endpoint, p256dh, auth from public.push_subscriptions where user_id = $1',
    [userId],
  )
  result.attempted = rows.length
  const body = JSON.stringify(payload)
  await Promise.all(
    rows.map(async (sub) => {
      try {
        await webpush.sendNotification(
          { endpoint: sub.endpoint, keys: { p256dh: sub.p256dh, auth: sub.auth } },
          body,
        )
        result.sent += 1
        await pool.query('update public.push_subscriptions set last_used_at = now() where id = $1', [sub.id])
      } catch (err) {
        const status = (err as { statusCode?: number })?.statusCode
        if (status === 404 || status === 410) {
          await pool.query('delete from public.push_subscriptions where id = $1', [sub.id])
          result.removed += 1
        } else {
          result.errors.push(`${sub.endpoint.slice(0, 40)}…: ${(err as Error).message}`)
        }
      }
    }),
  )
  return result
}

export function vapidPublicKey(): string | null {
  return PUBLIC ?? null
}
```

import { Hono } from 'hono'
import { z } from 'zod'
import { pool } from '../../db.js'
import { getIdentity } from '../../_shared/identity.js'
import { vapidPublicKey } from '../../_shared/webpush.js'
import { sendPush } from '../../_shared/webpush.js'
import type { AppVariables } from '../../types.js'

const SubscribeBody = z.object({
  endpoint: z.string().url(),
  keys: z.object({ p256dh: z.string(), auth: z.string() }),
  userAgent: z.string().optional(),
})

const UnsubscribeBody = z.object({ endpoint: z.string().url() })

export const push = new Hono<{ Variables: AppVariables }>()

push.get('/vapid-public-key', (c) => {
  return c.json({ key: vapidPublicKey() })
})

push.post('/subscribe', async (c) => {
  const json = await c.req.json().catch(() => null)
  const parsed = SubscribeBody.safeParse(json)
  if (!parsed.success) return c.json({ error: 'invalid body' }, 400)
  const { endpoint, keys, userAgent } = parsed.data
  const userId = getIdentity(c).userId
  await pool.query(
    `insert into public.push_subscriptions (user_id, endpoint, p256dh, auth, user_agent)
     values ($1, $2, $3, $4, $5)
     on conflict (user_id, endpoint)
     do update set p256dh = excluded.p256dh, auth = excluded.auth, user_agent = excluded.user_agent`,
    [userId, endpoint, keys.p256dh, keys.auth, userAgent ?? null],
  )
  return c.json({ ok: true }, 201)
})

push.delete('/subscribe', async (c) => {
  const json = await c.req.json().catch(() => null)
  const parsed = UnsubscribeBody.safeParse(json)
  if (!parsed.success) return c.json({ error: 'invalid body' }, 400)
  const userId = getIdentity(c).userId
  await pool.query(
    'delete from public.push_subscriptions where user_id = $1 and endpoint = $2',
    [userId, parsed.data.endpoint],
  )
  return c.body(null, 204)
})

push.post('/test', async (c) => {
  const userId = getIdentity(c).userId
  const result = await sendPush(userId, {
    title: 'Plannen',
    body: 'Push notifications are working.',
    url: '/dashboard',
    tag: 'plannen-test',

function urlBase64ToUint8Array(base64String: string): Uint8Array {
  const padding = '='.repeat((4 - (base64String.length % 4)) % 4)
  const base64 = (base64String + padding).replace(/-/g, '+').replace(/_/g, '/')
  const raw = atob(base64)
  const out = new Uint8Array(raw.length)
  for (let i = 0; i < raw.length; i += 1) out[i] = raw.charCodeAt(i)
  return out
}

export function pushSupported(): boolean {
  return typeof window !== 'undefined' &&
    'serviceWorker' in navigator &&
    'PushManager' in window &&
    'Notification' in window
}

export async function getPushSubscription(): Promise<PushSubscription | null> {
  if (!pushSupported()) return null
  const reg = await navigator.serviceWorker.ready
  return reg.pushManager.getSubscription()
}

export async function enablePush(): Promise<{ ok: boolean; reason?: string }> {
  if (!pushSupported()) return { ok: false, reason: 'unsupported' }
  const perm = await Notification.requestPermission()
  if (perm !== 'granted') return { ok: false, reason: 'denied' }

  const keyRes = await fetch(`${backendUrl()}/api/push/vapid-public-key`, { credentials: 'include' })
  const { key } = (await keyRes.json()) as { key: string | null }
  if (!key) return { ok: false, reason: 'server-not-configured' }

  const reg = await navigator.serviceWorker.ready
  let sub = await reg.pushManager.getSubscription()
  if (!sub) {
    sub = await reg.pushManager.subscribe({
      userVisibleOnly: true,
      applicationServerKey: urlBase64ToUint8Array(key),
    })
  }

  const json = sub.toJSON()
  const subscribeRes = await fetch(`${backendUrl()}/api/push/subscribe`, {
    method: 'POST',
    credentials: 'include',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({
      endpoint: json.endpoint,
      keys: json.keys,
      userAgent: navigator.userAgent,
    }),
  })
  if (!subscribeRes.ok) return { ok: false, reason: `server-${subscribeRes.status}` }
  return { ok: true }
}

export async function disablePush(): Promise<void> {
  const sub = await getPushSubscription()
  if (!sub) return
  await fetch(`${backendUrl()}/api/push/subscribe`, {
    method: 'DELETE',
    credentials: 'include',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({ endpoint: sub.endpoint }),
  })
  await sub.unsubscribe()
}

export async function sendTestPush(): Promise<boolean> {
  const res = await fetch(`${backendUrl()}/api/push/test`, {
    method: 'POST',
    credentials: 'include',
  })
  return res.ok
}

```tsx
import { useEffect, useState } from 'react'

type BeforeInstallPromptEvent = Event & {
  prompt: () => Promise<void>
  userChoice: Promise<{ outcome: 'accepted' | 'dismissed' }>
}

const DISMISS_KEY = 'plannen.install_dismissed_at'
const DISMISS_DAYS = 14

function dismissedRecently(): boolean {
  const v = localStorage.getItem(DISMISS_KEY)
  if (!v) return false
  const when = Number(v)
  if (!Number.isFinite(when)) return false
  return Date.now() - when < DISMISS_DAYS * 24 * 60 * 60 * 1000
}

function isIos(): boolean {
  return /iphone|ipad|ipod/i.test(navigator.userAgent) && !('MSStream' in window)
}

function isStandalone(): boolean {
  return window.matchMedia('(display-mode: standalone)').matches ||
    (navigator as Navigator & { standalone?: boolean }).standalone === true
}

export function InstallPrompt() {
  const [deferred, setDeferred] = useState<BeforeInstallPromptEvent | null>(null)
  const [showIos, setShowIos] = useState(false)

  useEffect(() => {
    if (isStandalone()) return
    if (dismissedRecently()) return

    const handler = (e: Event) => {
      e.preventDefault()
      setDeferred(e as BeforeInstallPromptEvent)
    }
    window.addEventListener('beforeinstallprompt', handler)

    if (isIos()) setShowIos(true)

    return () => window.removeEventListener('beforeinstallprompt', handler)
  }, [])

  const dismiss = () => {
    localStorage.setItem(DISMISS_KEY, String(Date.now()))
    setDeferred(null)
    setShowIos(false)
  }

  if (deferred) {
    return (
      <div className="fixed bottom-4 inset-x-4 z-50 rounded-lg bg-white shadow-lg border border-gray-200 p-3 flex items-center gap-3">
        <div className="flex-1 text-sm">
          <p className="font-medium text-gray-900">Install Plannen</p>
          <p className="text-gray-600">Use it like an app — faster, offline-ready.</p>
        </div>
        <button onClick={dismiss} className="text-xs text-gray-500 px-2 py-1">Later</button>
        <button
          onClick={async () => {
            await deferred.prompt()
            const choice = await deferred.userChoice
            if (choice.outcome) dismiss()
          }}
          className="px-3 py-1.5 rounded bg-indigo-600 text-white text-sm"
        >
          Install
        </button>
      </div>
    )
  }

  if (showIos) {
    return (
      <div className="fixed bottom-4 inset-x-4 z-50 rounded-lg bg-white shadow-lg border border-gray-200 p-3 flex items-start gap-3">
        <div className="flex-1 text-sm">
          <p className="font-medium text-gray-900">Add Plannen to your home screen</p>
          <p className="text-gray-600 mt-1">Tap the Share icon, then "Add to Home Screen".</p>
        </div>
        <button onClick={dismiss} className="text-xs text-gray-500 px-2 py-1">Got it</button>
      </div>
    )
  }

  return null
}
```
