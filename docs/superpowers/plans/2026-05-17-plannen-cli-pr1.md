# Plannen CLI — PR1: scaffold + 4 headline verbs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a Node CLI at `npx plannen` that wraps `bootstrap.sh`, `start.sh`, `stop.sh` behind verbs `init`, `up`, `down`, and adds a new native `status` verb — without changing any existing bash behaviour.

**Architecture:** A `bin/plannen.mjs` entry point uses [`citty`](https://github.com/unjs/citty) to parse argv and dispatch to per-verb modules under `cli/commands/`. Wrap verbs delegate to bash via a single `runScript` helper that injects env and propagates exit codes; the test for each wrap verb spies on a swappable spawner injected into `runScript`. The `status` verb is native (port-pings the conventional dev ports).

**Tech Stack:** Node 20+ ESM (`.mjs`), citty for argv parsing, vitest 4.x for tests (node environment, separate config from the web app's jsdom config), bash for the wrapped scripts (untouched).

```js
import { spawn as nodeSpawn } from 'node:child_process';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const __filename = fileURLToPath(import.meta.url);
const REPO_ROOT = path.resolve(path.dirname(__filename), '..', '..');

/**
 * Spawn `bash <repo>/<script>` with merged env. Resolves with the exit code.
 * Inherits stdio so child output reaches the user terminal directly.
 *
 * @param {object} opts
 * @param {string} opts.script   Path relative to repo root, e.g. 'scripts/start.sh'.
 * @param {string[]} [opts.args] Arguments passed to the script.
 * @param {Record<string,string>} [opts.env] Extra env vars merged onto process.env.
 * @param {Function} [opts.spawner] Override for tests; defaults to node:child_process spawn.
 * @returns {Promise<number>} Exit code (0 = success).
 */
export function runScript({ script, args = [], env = {}, spawner = nodeSpawn } = {}) {
  const absScript = path.isAbsolute(script) ? script : path.join(REPO_ROOT, script);
  const childEnv = { ...process.env, ...env };
  const child = spawner('bash', [absScript, ...args], {
    stdio: 'inherit',
    env: childEnv,
  });
  return new Promise((resolve) => {
    child.on('exit', (code, signal) => {
      if (signal) {
        // POSIX convention: signal-terminated processes exit with 128 + signal number.
        const sigNum = { SIGINT: 2, SIGTERM: 15, SIGHUP: 1, SIGKILL: 9 }[signal] ?? 1;
        resolve(128 + sigNum);
      } else {
        resolve(code ?? 1);
      }
    });
  });
}
```

```js
import { defineCommand } from 'citty';
import { runScript } from '../lib/runScript.mjs';

const MODE_TO_TIER = {
  local_pg: '0',
  local_sb: '1',
  cloud_sb: '2',
  tier0: '0',
  tier1: '1',
  tier2: '2',
};

/**
 * Pure entry usable from tests. citty calls `run()` (which calls this with
 * citty's parsed args). Tests can call invokeInit directly with a spawner.
 * Reads both kebab and camelCase keys — different citty versions normalize
 * differently and we don't want to be brittle.
 */
export async function invokeInit(rawArgs, { spawner } = {}) {
  const mode = rawArgs.mode;
  if (!mode || !(mode in MODE_TO_TIER)) {
    throw new Error(`init: --mode is required and must be one of: ${Object.keys(MODE_TO_TIER).join(', ')}`);
  }
  const nonInteractive = rawArgs['non-interactive'] ?? rawArgs.nonInteractive;
  const passthrough = ['--tier', MODE_TO_TIER[mode]];
  if (rawArgs.email) passthrough.push('--email', rawArgs.email);
  if (nonInteractive) passthrough.push('--non-interactive');

  return runScript({
    script: 'scripts/bootstrap.sh',
    args: passthrough,
    env: {},
    spawner,
  });
}

export const initCommand = defineCommand({
  meta: { name: 'init', description: 'Bootstrap Plannen (wraps scripts/bootstrap.sh)' },
  args: {
    mode: { type: 'string', description: 'Deployment mode: local_pg | local_sb | cloud_sb (or tier0|tier1|tier2)', required: true },
    email: { type: 'string', description: 'User email for the auth user row' },
    'non-interactive': { type: 'boolean', description: 'Fail rather than prompt' },
  },
  async run({ args }) {
    const code = await invokeInit(args);
    process.exit(code);
  },
});
```

import net from 'node:net';

/**
 * Resolves true if a TCP connect succeeds within `timeoutMs`, false otherwise.
 * Used by `plannen status` to detect whether each conventional dev port is live.
 *
 * @param {string} host
 * @param {number} port
 * @param {number} timeoutMs
 * @returns {Promise<boolean>}
 */
export function isPortOpen(host, port, timeoutMs = 250) {
  return new Promise((resolve) => {
    const socket = new net.Socket();
    let settled = false;
    const finish = (ok) => {
      if (settled) return;
      settled = true;
      socket.destroy();
      resolve(ok);
    };
    socket.setTimeout(timeoutMs);
    socket.once('connect', () => finish(true));
    socket.once('timeout', () => finish(false));
    socket.once('error', () => finish(false));
    socket.connect(port, host);
  });
}
```

import { defineCommand } from 'citty';
import { isPortOpen as defaultProbe } from '../lib/ports.mjs';

const PROCESSES_BY_TIER = {
  '0': [
    { name: 'pg', host: '127.0.0.1', port: 54322 },
    { name: 'backend', host: '127.0.0.1', port: 54323 },
    { name: 'web', host: '127.0.0.1', port: 4321 },
  ],
  '1': [
    { name: 'supabase', host: '127.0.0.1', port: 54321 },
    { name: 'pg', host: '127.0.0.1', port: 54322 },
    { name: 'web', host: '127.0.0.1', port: 4321 },
  ],
  '2': [
    { name: 'web', host: '127.0.0.1', port: 4321 },
  ],
};

export async function invokeStatus(rawArgs, ctx = {}) {
  const env = ctx.env ?? process.env;
  const probe = ctx.probe ?? defaultProbe;
  const out = ctx.out ?? process.stdout;
  const tier = String(env.PLANNEN_TIER ?? '0');
  const procs = PROCESSES_BY_TIER[tier] ?? PROCESSES_BY_TIER['0'];
  const results = await Promise.all(
    procs.map(async (p) => ({ ...p, up: await probe(p.host, p.port, 250) })),
  );

  if (rawArgs.json) {
    out.write(JSON.stringify({ tier, processes: results }, null, 2));
  } else {
    for (const r of results) {
      out.write(`${r.name.padEnd(10)} ${r.up ? 'up  ' : 'down'} :${r.port}\n`);
    }
  }
  return 0;
}

export const statusCommand = defineCommand({
  meta: { name: 'status', description: 'Show what is running for the active profile' },
  args: {
    json: { type: 'boolean', description: 'Emit JSON instead of human-readable lines' },
  },
  async run({ args }) {
    const code = await invokeStatus(args);
    process.exit(code);
  },
});
```
